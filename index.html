function renderMain() { const y = document.getElementById('filter-year').value; const q = document.getElementById('filter-quarter').value; // STEP 1: STRICT FILTERING const filtered = rawData.filter(r => { // Kunin ang data at linisin (remove spaces) const officeName = (r.OFFICE || "").toString().trim(); const accRate = (r['ACCOMPLISHMENT RATE'] || "").toString().trim(); const yearMatch = (r.YEAR || "").toString().trim() === y; const quarterMatch = (r.QUARTER || "").toString().trim().toUpperCase() === q.toUpperCase(); // LOGIC: Isasama LANG kung may pangalan ang Office AT may laman ang Accomplishment Rate // Kapag ang row ay mukhang may kulay pero walang text, dito siya mafi-filter out. return officeName !== "" && accRate !== "" && yearMatch && quarterMatch; }); // STEP 2: CHECK KUNG MAY DATA NA NATIRA if (filtered.length === 0) { console.warn("No valid data found for:", y, q); clearAll(); return; } // STEP 3: ENCODED CHECK (Dito natin tinitingnan kung kailan na-submit) const enc = filtered.filter(r => { const k = Object.keys(r).find(key => key.toUpperCase().includes('ENCODED')); return r[k] && r[k].toString().trim() !== ""; }); // STEP 4: MATH LOGIC const totalKPI = filtered.length; // Eto dapat ang magiging 14 const encodedCount = enc.length; const onTime = enc.filter(r => (r['WITHIN DEADLINE'] || "").toString().trim() === 'On Time').length; const late = enc.filter(r => (r['WITHIN DEADLINE'] || "").toString().trim() === 'Late').length; let sum = 0; enc.forEach(r => { const val = (r['ACCOMPLISHMENT RATE'] || "0").toString().replace('%', ''); sum += parseFloat(val) || 0; }); // STEP 5: DISPLAY SA DASHBOARD (ID check) const updateText = (id, text) => { const el = document.getElementById(id); if (el) el.innerText = text; }; updateText('stat-total', totalKPI); updateText('stat-encoded', encodedCount); updateText('stat-avg', (encodedCount ? (sum / encodedCount).toFixed(2) : "0.00") + "%"); updateText('stat-comp', (totalKPI ? (encodedCount / totalKPI * 100).toFixed(2) : "0.00") + "%"); updateText('stat-ontime', onTime); updateText('stat-late', late); updateText('stat-missing', totalKPI - encodedCount); // STEP 6: REFRESH CHARTS const group = {}; filtered.forEach(r => { const off = r.OFFICE.toString().trim(); const rate = parseFloat((r['ACCOMPLISHMENT RATE'] || "0").toString().replace('%', '')) || 0; group[off] = rate; }); if (typeof drawMainCharts === "function") { drawMainCharts(Object.keys(group), Object.values(group), encodedCount, totalKPI - encodedCount); } if (typeof buildRank === "function") { buildRank(enc); } }
