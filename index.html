function renderMain() { const y = document.getElementById('filter-year').value; const q = document.getElementById('filter-quarter').value; // 1. KUNIN ANG MGA KPI NA MAY LAMAN ANG ACCOMPLISHMENT (Ito ang magiging Base/Total) const filteredWithData = rawData.filter(r => { const hasOffice = (r.OFFICE || "").trim() !== ""; const hasAccRate = (r['ACCOMPLISHMENT RATE'] || "").trim() !== ""; // Dapat may data ang accomplishment const matchYear = (r.YEAR || "").trim() === y; const matchQuarter = (r.QUARTER || "").trim().toUpperCase() === q.toUpperCase(); return hasOffice && hasAccRate && matchYear && matchQuarter; }); // 2. KUNIN KUNG ALIN SA MGA IYAN ANG MAY DATA NA SA ENCODED const encodedWithData = filteredWithData.filter(r => { // Hanapin ang column na may "ENCODED" sa pangalan const encodedKey = Object.keys(r).find(key => key.toUpperCase().includes('ENCODED')); const hasEncodedDate = r[encodedKey] && r[encodedKey].toString().trim() !== ""; return hasEncodedDate; }); if (filteredWithData.length === 0) { clearAll(); return; } // 3. MGA BILANG PARA SA DASHBOARD const totalKPI = filteredWithData.length; // Eto yung 14 mo const encodedCount = encodedWithData.length; // Within Deadline Logic (Base lang sa mga na-encode na) const onTime = encodedWithData.filter(r => (r['WITHIN DEADLINE'] || "").trim() === 'On Time').length; const late = encodedWithData.filter(r => (r['WITHIN DEADLINE'] || "").trim() === 'Late').length; // Accomplishment Rate Average (Base sa 14 na may data) let totalSum = 0; filteredWithData.forEach(r => { const val = (r['ACCOMPLISHMENT RATE'] || "0").toString().replace('%', ''); totalSum += parseFloat(val) || 0; }); // 4. DISPLAY SA UI document.getElementById('stat-total').innerText = totalKPI; document.getElementById('stat-encoded').innerText = encodedCount; document.getElementById('stat-avg').innerText = (totalKPI ? (totalSum / totalKPI).toFixed(2) : "0.00") + "%"; document.getElementById('stat-comp').innerText = (totalKPI ? (encodedCount / totalKPI * 100).toFixed(2) : "0.00") + "%"; document.getElementById('stat-ontime').innerText = onTime; document.getElementById('stat-late').innerText = late; document.getElementById('stat-missing').innerText = totalKPI - encodedCount; // Update Charts const group = {}; filteredWithData.forEach(r => { const off = r.OFFICE.trim(); const rate = parseFloat((r['ACCOMPLISHMENT RATE'] || "0").toString().replace('%', '')) || 0; group[off] = rate; }); drawMainCharts(Object.keys(group), Object.values(group), encodedCount, totalKPI - encodedCount); }
