function renderMain() { const y = document.getElementById('filter-year').value, q = document.getElementById('filter-quarter').value; // STRICT FILTERING: Dapat may Office AND may Accomplishment data const filtered = rawData.filter(r => { const officeName = (r.OFFICE || "").trim(); const accRate = (r['ACCOMPLISHMENT RATE'] || "").trim(); const yearMatch = (r.YEAR || "").trim() === y; const quarterMatch = (r.QUARTER || "").trim().toUpperCase() === q; // HINDI isasali kung blanko ang Office o blanko ang Accomplishment Rate return officeName !== "" && accRate !== "" && yearMatch && quarterMatch; }); const enc = filtered.filter(r => { const k = Object.keys(r).find(key => key.toUpperCase().includes('ENCODED')); return r[k] && r[k].trim() !== ""; }); if(!filtered.length) { clearAll(); return; } // Math Logic const totalKPI = filtered.length; // Dito lalabas ang 14 base sa may data lang const encodedCount = enc.length; const onTime = enc.filter(r => (r['WITHIN DEADLINE'] || "").trim() === 'On Time').length; const late = enc.filter(r => (r['WITHIN DEADLINE'] || "").trim() === 'Late').length; let sum = 0; enc.forEach(r => sum += parseFloat((r['ACCOMPLISHMENT RATE'] || "0").replace('%', '')) || 0); // Display sa Dashboard document.getElementById('stat-total').innerText = totalKPI; document.getElementById('stat-encoded').innerText = encodedCount; document.getElementById('stat-avg').innerText = (encodedCount ? (sum / encodedCount).toFixed(2) : "0.00") + "%"; document.getElementById('stat-comp').innerText = (totalKPI ? (encodedCount / totalKPI * 100).toFixed(2) : "0.00") + "%"; document.getElementById('stat-ontime').innerText = onTime; document.getElementById('stat-late').innerText = late; document.getElementById('stat-missing').innerText = totalKPI - encodedCount; // Update Charts const group = {}; filtered.forEach(r => { const off = r.OFFICE.trim(); const rate = parseFloat((r['ACCOMPLISHMENT RATE'] || "0").replace('%', '')) || 0; group[off] = rate; }); drawMainCharts(Object.keys(group), Object.values(group), encodedCount, totalKPI - encodedCount); buildRank(enc); }
